1) Conta gli ospiti raggruppandoli per anno di nascita

	SELECT YEAR(`date_of_birth`) AS BirthDate, COUNT(id) AS YearCount FROM `ospiti` GROUP BY YEAR(`date_of_birth`);



2) Somma i prezzi dei pagamenti raggruppandoli per status

	SELECT SUM(`price`) AS SOMMA, `status` AS STATUS FROM `pagamenti` GROUP BY `status`;
